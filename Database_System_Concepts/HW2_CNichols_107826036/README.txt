CREATE TABLE CUSTOMER (
    IDNo		INT	NOT NULL,
    FInit		CHAR(1)	Default NULL,
    LName		VARCHAR(12)	Default NULL,
    Phone		VARCHAR(12) DEFAULT "000-000-0000",
    PRIMARY KEY	(IDNo));

CREATE TABLE Cars (
    VehicleID		INT	NOT NULL,
    Available		BINARY	Default 0 NOT NULL,
    MakeYear		INT NOT NULL,
    Type			VARCHAR(12)	NOT NULL,
    Model			VARCHAR(7)	NOT NULL,
    DailyRate		FLOAT(4, 2) Default 0.00,
    WeeklyRate		FLOAT(4, 2) Default 0.00,
    PRIMARY KEY	(VehicleID));

CREATE TABLE Rental (
    VehicleID		INT	NOT NULL,
    IDNo			INT	NOT NULL,
    StartDate		DATE NOT NULL,
    Daily			BINARY	Default 0 NOT NULL,
    NumDays			INT Default 0 NOT NULL,
    ReturnDate		DATE DEFAULT NULL,
    PRIMARY KEY	(VehicleID, StartDate),
    FOREIGN KEY (VehicleID) REFERENCES Cars(VehicleID),
    FOREIGN KEY (IDNo) REFERENCES Customer(IDNo));
